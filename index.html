1
Открой таблицу Google Sheets → меню Расширения → Apps Script
2
Удали весь код и вставь:

function doPost(e){try{var d=JSON.parse(e.postData.contents);var ss=SpreadsheetApp.getActiveSpreadsheet();var sh=ss.getSheetByName('Statistik')||ss.insertSheet('Statistik');sh.clearContents();sh.appendRow(['Kapitel','Слово','Перевод','🇩🇪попыток','🇩🇪ошибок','🇷🇺попыток','🇷🇺ошибок','🔊попыток','🔊ошибок','🔤попыток','🔤ошибок','Всего ошибок','Обновлено']);d.rows.forEach(function(r){sh.appendRow(r);});return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);}catch(err){return ContentService.createTextOutput(JSON.stringify({ok:false,error:err.message})).setMimeType(ContentService.MimeType.JSON);}}
3
